import java.util.Scanner;

class Ecom 
{
String name;
double price;
int qty;

Ecom(String n, double p, int q) 
{
name = n;
price = p;
qty = q;
}

double total() 
{
return price * qty;
}

    
void display() 
{
System.out.println(name + " | Price: " + price +
                " | Quantity: " + qty +
                " | Total: " + total());
}

public static void main(String[] args) 
{

Scanner sc = new Scanner(System.in);

Ecom p1 = new Ecom("Powerbank", 800, 1);
Ecom p2 = new Ecom("Headphones", 150, 2);

System.out.print("Enter product name: ");
String name = sc.nextLine();

System.out.print("Enter product price: ");
double price = sc.nextDouble();

System.out.print("Enter quantity: ");
int qty = sc.nextInt();

Ecom p3 = new Ecom(name, price, qty);

       
System.out.println("\nOrder Summary:");
p1.display();
p2.display();
p3.display();

double total = p1.total() + p2.total() + p3.total();

double finalPrice;

if (total > 15000)
finalPrice = total - total * 0.25;
else if (total > 10000)
finalPrice = total - total * 0.20;
else if (total > 5000)
finalPrice = total - total * 0.15;
else
finalPrice = total;

System.out.println("\nTotal Cost Before Discount: " + total);
System.out.println("Final Price After Discount: " + finalPrice);
System.out.println("Thank you for shopping!");

sc.close();
}
}
